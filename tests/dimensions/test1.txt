# Determinar distanca percorrida por uma massa em t segundos
# Pedir ao user:
#     velocidade inicial
#     Força

# Resultado: distancia percorrida


#Time t = input("Insira o tempo") [s];
Time t = 16 [s];
Mass m = 1 [kg];
Force f = 20 [N];
Velocity v0 = 0 [m/s];

Accelaration a = f / m;

Distance res = v0 * t + (a*t^2)/2;

print("Distancia percorrida: "+res);
